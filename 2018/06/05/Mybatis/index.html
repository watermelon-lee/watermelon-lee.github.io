package com.mapper;

import com.domain.Student;

public interface StudentDao {
Student select(Integer id);
}

private SqlSession sqlSession;

@Test
public void OneToOne() {
    InputStream inputStream=this.getClass().getResourceAsStream("mybatis/Mybatis-config.xml");//从target中的classes文件夹中读取文件

    //初始化mybatis,创建SqlsessionFactory实例
    private SqlSessionFactory sqlSessionFactory=new SqlSessionFactoryBuilder().build(inputStream);
    //创建Session实例
    sqlSession = sqlSessionFactory.openSession();
    //获得mapper接口的代理
    StudentDao studentDao = sqlSession.getMapper(StudentDao.class);
    ClazzDao clazzDao=sqlSession.getMapper(ClazzDao.class);
    Clazz clazz=clazzDao.select(1);
    System.out.println(clazz);
    //直接调用接口查询
    Student student=studentDao.select(1);
    System.out.println(student);
    sqlSession.commit();
    sqlSession.close();
}

}
